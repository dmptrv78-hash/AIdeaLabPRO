# ---------- ВСПОМОГАТЕЛЬНАЯ СВОДКА ----------
async def show_summary(message: types.Message, state: FSMContext, service_name: str, fields: dict):
    data = await state.get_data()
    summary = f"📋 {service_name}\n\n"
    for key, label in fields.items():
        val = data.get(key, '—')
        summary += f"{label}: {val}\n"
    await message.answer(
        f"{summary}\n\n✅ Заявка создана! Передаём специалисту.",
        reply_markup=main_menu_keyboard()
    )
    await state.clear()

@dp.message(lambda msg: msg.text == "🔙 Назад")
async def go_back(message: types.Message, state: FSMContext):
    current_state = await state.get_state()
    if not current_state:
        await message.answer("Вы не в процессе заполнения.", reply_markup=main_menu_keyboard())
        return

    # Общий словарь переходов для всех услуг
    back_map = {
        "TZStates": {
            "essence": "name", "audience": "essence", "features": "audience",
            "competitors": "features", "tech_limits": "competitors",
            "deadline": "tech_limits", "budget": "deadline", "files": "budget"
        },
        "TEOStates": {
            "resources": "goal", "risks": "resources", "norms": "risks",
            "effect": "norms", "data": "effect", "horizon": "data", "files": "horizon"
        },
        "FMStates": {
            "costs": "income", "investment": "costs", "breakeven": "investment",
            "metrics": "breakeven", "data": "metrics", "horizon": "data"
        },
        "BPStates": {
            "product": "summary", "competitors": "product", "marketing": "competitors",
            "team": "marketing", "sales": "team", "risks": "sales",
            "capital": "risks", "finance_file": "capital", "files": "finance_file"
        },
        "ConsultStates": {
            "stage": "description", "goal": "stage"
        }
    }

    prefix = current_state.split(":")[0]
    step = current_state.split(":")[1]
    if prefix in back_map and step in back_map[prefix]:
        prev = back_map[prefix][step]
        # Динамически получаем класс состояний
        cls = globals()[prefix]
        await state.set_state(getattr(cls, prev))
        await message.answer("Вернулись к предыдущему вопросу.", reply_markup=nav_keyboard())
    else:
        await message.answer("Это первый шаг, назад нельзя.", reply_markup=nav_keyboard())

@dp.message(ConsultStates.stage)
async def process_consult_stage(message: types.Message, state: FSMContext):
    if message.text not in ["💡 Идея", "⚙️ Прототип", "🚀 Готовый продукт"]:
        await message.answer("Пожалуйста, выберите из предложенных вариантов.", reply_markup=stage_keyboard())
        return
    await state.update_data(stage=message.text)
    await state.set_state(ConsultStates.goal)
    await message.answer(
        "Чего вы хотите достичь с нашей помощью?",
        reply_markup=nav_keyboard()
    )

@dp.message(ConsultStates.goal)
async def process_consult_goal(message: types.Message, state: FSMContext):
    if not message.text or len(message.text.strip()) == 0:
        await message.answer("Пожалуйста, опишите цель.")
        return
    await state.update_data(goal=message.text)
    
    # Рекомендация на основе этапа
    data = await state.get_data()
    stage = data.get("stage", "")
    if "Идея" in stage:
        recommendation = "На основе вашего этапа мы рекомендуем начать с **Технического задания**."
    elif "Прототип" in stage:
        recommendation = "У вас уже есть прототип — возможно, стоит подготовить **Бизнес-план** для привлечения инвестиций."
    else:
        recommendation = "Для готового продукта мы рекомендуем **Финансовую модель** и **ТЭО** для масштабирования."
    
    fields = {
        "description": "Проект",
        "stage": "Этап",
        "goal": "Цель"
    }
    summary = f"📋 КОНСУЛЬТАЦИЯ\n\n"
    for key, label in fields.items():
        summary += f"{label}: {data.get(key, '—')}\n"
    summary += f"\n💡 {recommendation}\n\n"
    summary += "Хотите сразу перейти к оформлению заказа? Нажмите на нужную услугу в меню."
    
    await message.answer(summary, reply_markup=main_menu_keyboard())
    await state.clear()
