# ---------- ВСПОМОГАТЕЛЬНАЯ ФУНКЦИЯ ДЛЯ ОТОБРАЖЕНИЯ СВОДКИ ----------
async def show_summary(message: types.Message, state: FSMContext, service_name: str, fields: dict):
    data = await state.get_data()
    summary = f"📋 {service_name}\n\n"
    for key, label in fields.items():
        summary += f"{label}: {data.get(key, '—')}\n"
    await message.answer(
        f"{summary}\n\n✅ Заявка создана! Передаём специалисту.",
        reply_markup=main_menu_keyboard()
    )
    await state.clear()

@dp.message(lambda msg: msg.text == "🔙 Назад")
async def go_back(message: types.Message, state: FSMContext):
    # Получаем текущее состояние
    current_state = await state.get_state()
    if not current_state:
        await message.answer("Вы не в процессе заполнения.", reply_markup=main_menu_keyboard())
        return
    
    # Определяем, к какой группе состояний относится текущее
    if current_state.startswith("TZStates:"):
        # Возвращаемся на предыдущий шаг ТЗ
        step = current_state.split(":")[1]
        prev_steps = {
            "essence": "name",
            "audience": "essence",
            "features": "audience",
            "competitors": "features",
            "tech_limits": "competitors",
            "deadline": "tech_limits",
            "budget": "deadline",
            "files": "budget"
        }
        if step in prev_steps:
            await state.set_state(getattr(TZStates, prev_steps[step]))
            # Отправляем сообщение с предыдущим вопросом (можно вынести в отдельную функцию)
            await message.answer("Возврат на предыдущий шаг. Пожалуйста, введите данные заново.", reply_markup=nav_keyboard())
        else:
            await message.answer("Это первый шаг, назад нельзя.", reply_markup=nav_keyboard())
    elif current_state.startswith("TEOStates:"):
        # Аналогично для ТЭО
        step = current_state.split(":")[1]
        prev_steps_teo = {
            "resources": "goal",
            "risks": "resources",
            "norms": "risks",
            "effect": "norms",
            "data": "effect",
            "horizon": "data",
            "files": "horizon"
        }
        if step in prev_steps_teo:
            await state.set_state(getattr(TEOStates, prev_steps_teo[step]))
            await message.answer("Возврат на предыдущий шаг.", reply_markup=nav_keyboard())
        else:
            await message.answer("Это первый шаг, назад нельзя.", reply_markup=nav_keyboard())
    else:
        await message.answer("Неизвестное состояние.", reply_markup=main_menu_keyboard())
